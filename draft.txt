/app
├─ /api
│  ├─ /auth
│  │  ├─ signup/route.js         // Register user & send OTP
│  │  ├─ login/route.js          // Login user and return JWT
│  │  ├─ verify-otp/route.js     // Verify OTP
│  │  ├─ forgot-password/route.js
│  │  └─ reset-password/route.js
│  ├─ /products
│  │  ├─ route.js                // GET all products, POST new product
│  │  └─ [id]/route.js           // GET, PUT, DELETE product by ID
│  ├─ /vendors
│  │  ├─ route.js                // GET all vendors, create vendor
│  │  └─ [id]/route.js           // GET, PUT vendor by ID
│  ├─ /orders
│  │  ├─ route.js                // POST new order, GET all orders
│  │  └─ [id]/route.js           // GET, PUT, DELETE order by ID
│  ├─ /payments
│  │  └─ route.js                // Payment creation & verification
│  └─ /notifications
│     └─ email/route.js          // Send emails (verification, order, etc.)
